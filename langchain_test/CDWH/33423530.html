DataWarehouse : IL_DIM_ORG: Organisationseinheiten (OE)

Created by Nicole Goebel, last modified by Fabian Franzeck on Aug. 14, 2023

To get an overview of all organizational units, the disciplines and locations are listed here with their ID. 

The differentiation of discipline and location is done via a 'role playing BK'.

For full coverage of all movements (including entry) use cdwh.V_IL_FCT_MOVEMENT (first Codeblock).


--All sites & disciplines cdwh.V_IL_FCT_MOVEMENT				==> covers ALL movements (including entry)
----------------------------------------------------------------
SELECT DISTINCT 
	 a.MOV_ASSIGN_ORG_BK 			AS MOV_Standord_ID			--00000xxxx (4digits
	,b.org 							AS MOV_Standord
	,a.MOV_DEPT_ORG_BK 				AS MOV_Fachdisziplin_ID 	--000000xxx (3digits)
	,b2.org 						AS MOV_Fachdisziplin
FROM	cdwh.V_IL_FCT_MOVEMENT 		AS a
JOIN 	cdwh.V_IL_DIM_ORG_CID 		AS b
ON 		a.MOV_ASSIGN_ORG_BK = b.org_bk 			-- = 'role playing bk': Schlüsselt Org auf zu: Standord (Notfallzentrum)
JOIN 	cdwh.V_IL_DIM_ORG_CID 		AS b2
ON 		a.MOV_DEPT_ORG_BK = b2.org_bk 			-- = 'role playing bk': schlüsselt Org auf zu: Fachdisziplin (Notfall Medizin), wenn man das auch möchte
ORDER BY MOV_Fachdisziplin, MOV_Standord
--All sites & disciplines from cdwh.V_IL_DIM_CASE_CID			==> are already included in the Movements (redundancy!)
----------------------------------------------------------------
SELECT DISTINCT 
	 a.CAS_ADMISSION_assign_ORG_BK 	AS CAS_Standord_ID			--00000xxxx (4stellig)
	,b.org 							AS CAS_Standord
	,a.CAS_ADMISSION_dept_ORG_BK 	AS CAS_Fachdisziplin_ID 	--000000xxx (3stellig)
	,b2.org 						AS CAS_Fachdisziplin
--	,a.*
FROM	cdwh.V_IL_DIM_CASE_CUR 			AS a
JOIN 	cdwh.V_IL_DIM_ORG_CID 			AS b
ON 		a.CAS_ADMISSION_assign_ORG_BK = b.org_bk 	-- = 'role playing bk': Schlüsselt Org auf zu: Standord (Notfallzentrum)
JOIN 	cdwh.V_IL_DIM_ORG_CID 			AS b2
ON 		a.CAS_ADMISSION_dept_ORG_BK = b2.org_bk 	-- = 'role playing bk': schlüsselt Org auf zu: Fachdisziplin (Notfall Medizin), wenn man das auch möchte
ORDER BY CAS_Fachdisziplin, CAS_Standord
--Discharge department from cdwh.V_IL_DIM_ORG_CID
----------------------------------------------------------------
SELECT	CAS_DISCHARGE_DEPT_ORG_BK, b.ORG, count(*)
FROM	cdwh.V_IL_DIM_CASE_CUR AS c
JOIN 	cdwh.V_IL_DIM_ORG_CID AS b ON	c.CAS_DISCHARGE_DEPT_ORG_BK = b.ORG_BK
WHERE	1=1
	AND CAS_ADMISSION_TYPE_ID <> 'EA'
	AND CAS_TYPE = 'stationär'
--	AND CAS_DISCHARGE_DEPT_ORG_BK IN ('00000320' , '00000337' , '00000328' , '00000332' , '00000373' , '00000326' , '00000331' , '00000323' , '00000330' , '00000393' , '00000324', '00000336')
GROUP BY
	CAS_DISCHARGE_DEPT_ORG_BK
	, b.ORG
